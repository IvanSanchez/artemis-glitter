
Known data types and corresponding buffer unpacking functions:

byte
buffer.readUInt8()

short
buffer.readUInt16LE()

int
buffer.readUInt32LE()

float
buffer.readFloatLE()


string
var strLen    = data.readUInt32LE(start) * 2;
var name      = data.toString('utf16le', (start+4), (start+4) +strLen-2);
data = data.slice((start+4) + strLen );
